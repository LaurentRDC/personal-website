-- Haskell
fib :: Integer -> Integer
fib 0 = 0
fib 1 = 1
fib n = fib (n - 1) + fib (n - 2)
# Python
def fib(n):
    i, j = 0, 1
    for _ in range(n):
        yield i
        i, j = j, i + j
